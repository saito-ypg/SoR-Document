218XMVECTOR Player::getMouseTargetPos()
219{
220 XMFLOAT3 mouse = Input::GetMousePosition();
221 XMMATRIX matInv = Camera::GetInverseMatrix();
222 XMFLOAT3 front = mouse, back = mouse;
223 back.z = 1.0f;
224 XMVECTOR vFront = XMVector3TransformCoord(XMLoadFloat3(&front), matInv);
225 XMVECTOR vBack = XMVector3TransformCoord(XMLoadFloat3(&back), matInv);
226 XMVECTOR dir = XMVector3Normalize(vBack - vFront);
227 XMVECTOR a = dir / XMVectorGetY(dir);//yが1になるように
