119GameObject * GameObject::FindChildObject(const std::string & name)
120{
121 //子供がいないなら終わり
122 if (childList_.empty())
123 return nullptr;
124
125 //イテレータ
126 auto it = childList_.begin(); //先頭
127 auto end = childList_.end(); //末尾
128
129 //子オブジェクトから探す
130 while (it != end) {
131 //同じ名前のオブジェクトを見つけたらそれを返す
132 if ((*it)->GetObjectName() == name)
133 return *it;
134
135 //その子供（孫）以降にいないか探す
136 GameObject* obj = (*it)->FindChildObject(name);
137 if (obj != nullptr)
138 {
139 return obj;
140 }
141
142 //次の子へ
143 it++;
144 }
145
146 //見つからなかった
147 return nullptr;

235//衝突判定
236void GameObject::Collision(GameObject * pTarget)
237{
238 //自分同士の当たり判定はしない
239 if (pTarget == nullptr || this == pTarget)
240 {
241 return;
242 }
243
244 //自分とpTargetのコリジョン情報を使って当たり判定
245 //1つのオブジェクトが複数のコリジョン情報を持ってる場合もあるので二重ループ
246 for (auto i = this->colliderList_.begin(); i != this->colliderList_.end(); i++)
247 {
248 for (auto j = pTarget->colliderList_.begin(); j != pTarget->colliderList_.end(); j++)
249 {
250 if ((*i)->IsHit(*j))
251 {
252 //当たった
253 this->OnCollision(pTarget);
254 }
255 }
256 }
257
258 //子供がいないなら終わり
259 if (pTarget->childList_.empty())
260 return;
261
262 //子供も当たり判定
263 for (auto i = pTarget->childList_.begin(); i != pTarget->childList_.end(); i++)
264 {
265 Collision(*i);
266 }
267}
