47
48 //スプライトはPositionを1ずらすと画面サイズの半分ずれるので、ピクセル単位に変換
49 px /= (float)(Direct3D::screenWidth_ / 2.0f);
50 py /= (float)(Direct3D::screenHeight_ / 2.0f);
51
52
53 //１文字ずつ表示する
54 for (int i = 0; str[i] != '\0'; i++) //文字列の末尾まで来たら終わり

56 //表示したい文字が、画像の何番目に書いてあるかを求める

58

60 int x_ = id % rowLength_; //左から何番目
61 int y_ = id / rowLength_; //上から何番目
62
