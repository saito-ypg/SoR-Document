58

60 Input::Initialize(hWnd);
61
62 //オーディオ（効果音）の準備
63 Audio::Initialize();
64
65
66 //ルートオブジェクト準備
67 //すべてのゲームオブジェクトの親となるオブジェクト
68 RootObject* pRootObject = new RootObject(nullptr);
69 pRootObject->Initialize();
70
71
72 //メッセージループ（何か起きるのを待つ）
73 MSG msg;
74 ZeroMemory(&msg, sizeof(msg));
75 while (msg.message != WM_QUIT)
76 {
77 //メッセージあり（こっちが優先）
78 if (PeekMessage(&msg, nullptr, 0U, 0U, PM_REMOVE))
79 {
80 TranslateMessage(&msg);
81 DispatchMessage(&msg);
82 }
83
84 //メッセージなし（ここでゲームの処理）
85 else
86 {
87 //時間計測
88 timeBeginPeriod(1); //時間計測の制度を上げる
89 static int FPS = 0; //画面更新回数のカウンタ
90 static DWORD lastFpsResetTime = timeGetTime(); //最後にキャプションにFPSを表示した
91 static DWORD lastUpdateTime = timeGetTime(); //最後に画面を更新した時間
92 DWORD nowTime = timeGetTime(); //現在の時間
93
94 //キャプションに現在のFPSを表示する
95 if (isDrawFps)
96 {
97 //1秒（1000ミリ秒）経ったら
98 if (nowTime - lastFpsResetTime > 1000)
99 {
100 //FPSの値を表示
101 char string[16];
102 wsprintf(string, "FPS:%d", FPS);
103 SetWindowText(GetActiveWindow(), string);
104 FPS = 0;
105 lastFpsResetTime = nowTime;
106 }
107 }
108 DWORD diff = nowTime - lastUpdateTime;
109
110 //指定した時間（FPSを60に設定した場合は60分の1秒）経過していたら更新処理
111 if ((diff) * fpsLimit > 1000.0f)
112 {
113 //時間計測関連
114 lastUpdateTime = nowTime; //現在の時間（最後に画面を更新した時間）を覚えておく
115 FPS++; //画面更新回数をカウントする
116
117
118
119
120 //入力（キーボード、マウス、コントローラー）情報を更新
121 Input::Update();
122
123 //全オブジェクトの更新処理
124 //ルートオブジェクトのUpdateを呼んだあと、自動的に子、孫のUpdateが呼ばれる
125 pRootObject->UpdateSub(float(diff < DELTA_MAX ? diff : DELTA_MAX));
126
127 //カメラを更新
128 Camera::Update();
129
130 //エフェクトの更新
131 VFX::Update();
132
133
134 //このフレームの描画開始
135 Direct3D::BeginDraw();
136
137 //全オブジェクトを描画
138 //ルートオブジェクトのDrawを呼んだあと、自動的に子、孫のUpdateが呼ばれる
139 pRootObject->DrawSub();
140
141 //エフェクトの描画
142 VFX::Draw();
143
144 //描画終了
145 Direct3D::EndDraw();
146
147
148
149
150 //ちょっと休ませる
151 Sleep(1);
152 }
153 timeEndPeriod(1); //時間計測の制度を戻す
154 }
155 }
156
157
158 pRootObject->ReleaseSub();
159 //いろいろ解放
160 VFX::Release();
161 Audio::AllRelease();
162 Model::AllRelease();
163 Image::AllRelease();
164
165 SAFE_DELETE(pRootObject);
166 Direct3D::Release();
167 return 0;
168}
