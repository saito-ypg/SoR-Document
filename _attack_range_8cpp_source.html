110{

112 s.y = 0;

114 XMFLOAT3 actor= *data.pCollider->position_;
115 actor.y = 0;
116 XMVECTOR ActorPos = XMLoadFloat3(&actor);
117 if (XMVectorGetX(XMVector3Length(XMVectorAbs(sectorPos - ActorPos))) > radius_ + data.pActor->GetRadius())//まずは円と同じ
118 {
119 return false;
120 }
121 float radAngle = XMConvertToRadians(centerAngle_);//開き具合
122 float radRot = XMConvertToRadians(rotate_);//扇の回転
123 XMMATRIX rotM = XMMatrixRotationY(radRot);
124 XMVECTOR Front{ 0,0,1,0 };
125 XMVECTOR rotFront = XMVector3TransformCoord(Front, rotM);//ここで扇形の回転具合
126 //角度→(a・b)/|a||b|
127 XMVECTOR sectorToActor = ActorPos - sectorPos;
128 float deviation = XMVectorGetX(XMVector3AngleBetweenVectors(rotFront, sectorToActor));//ここで扇のどの辺にいるか角度が出せるはず
129 if (std::abs(deviation) > radAngle)
130 {//中心が扇の外なら追加で検証
131 XMMATRIX angleM = XMMatrixRotationY(radAngle);//開き具合
132 float a, b;
133 a = (std::abs(XMVectorGetX(XMVector3AngleBetweenVectors(XMVector3TransformCoord(rotFront, angleM), sectorToActor))));//片方の端
134 b = std::abs(XMVectorGetX(XMVector3AngleBetweenVectors(XMVector3TransformCoord(rotFront, XMMatrixInverse(nullptr, angleM)), sectorToActor)));//もう片方の端。回転の逆行列
135 float close = a < b?a:b;//両端を見て近いほうのrad

141 }
