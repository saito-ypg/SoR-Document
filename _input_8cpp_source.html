218 float GetAnalogValue(int raw, int max, int deadZone)
219 {
220 float result = (float)raw;
221
222 if (result > 0)
223 {
224 //デッドゾーン
225 if (result < deadZone)
226 {
227 result = 0;
228 }
229 else
230 {
231 result = (result - deadZone) / (max - deadZone);
232 }
233 }
234
235 else
236 {
237 //デッドゾーン
238 if (result > -deadZone)
239 {
240 result = 0;
241 }
242 else
243 {
244 result = (result + deadZone) / (max - deadZone);
245 }
246 }
247
248 return result;
249 }
