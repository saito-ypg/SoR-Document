72//頂点バッファ準備
73void FbxParts::InitVertex(fbxsdk::FbxMesh * mesh)
74{
75 pVertexData_ = new VERTEX[vertexCount_];
76
77 for (DWORD poly = 0; poly < polygonCount_; poly++)
78 {
79 //3頂点分
80 for (int vertex = 0; vertex < 3; vertex++)
81 {
82 int index = mesh->GetPolygonVertex(poly, vertex);
83
85 FbxVector4 pos = mesh->GetControlPointAt(index);
86 pVertexData_[index].position = XMFLOAT3((float)-pos[0], (float)pos[1], (float)pos[2]);
87
89 FbxVector4 Normal;
90 mesh->GetPolygonVertexNormal(poly, vertex, Normal); //ｉ番目のポリゴンの、ｊ番目の頂点の法線をゲット
91 pVertexData_[index].normal = XMFLOAT3((float)-Normal[0], (float)Normal[1], (float)Normal[2]);
92
94 FbxLayerElementUV * pUV = mesh->GetLayer(0)->GetUVs();
95 int uvIndex = mesh->GetTextureUVIndex(poly, vertex, FbxLayerElement::eTextureDiffuse);
96 FbxVector2 uv = pUV->GetDirectArray().GetAt(uvIndex);
97 pVertexData_[index].uv = XMFLOAT3((float)uv.mData[0], (float)(1.0f - uv.mData[1]), 0.0f);
98 }
99 }
100
101
103 int m_dwNumUV = mesh->GetTextureUVCount();
104 FbxLayerElementUV * pUV = mesh->GetLayer(0)->GetUVs();
105 if (m_dwNumUV > 0 && pUV->GetMappingMode() == FbxLayerElement::eByControlPoint)
106 {
107 for (int k = 0; k < m_dwNumUV; k++)
108 {
109 FbxVector2 uv = pUV->GetDirectArray().GetAt(k);
110 pVertexData_[k].uv = XMFLOAT3((float)uv.mData[0], (float)(1.0f - uv.mData[1]), 0.0f);
111 }
112 }
113
114
115 // 頂点データ用バッファの設定
116 D3D11_BUFFER_DESC bd_vertex;
117 bd_vertex.ByteWidth = sizeof(VERTEX) * mesh->GetControlPointsCount();
118 bd_vertex.Usage = D3D11_USAGE_DYNAMIC;
119 bd_vertex.BindFlags = D3D11_BIND_VERTEX_BUFFER;
120 bd_vertex.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
121 bd_vertex.MiscFlags = 0;
122 bd_vertex.StructureByteStride = 0;
123 D3D11_SUBRESOURCE_DATA data_vertex;
124 data_vertex.pSysMem = pVertexData_;
125 Direct3D::pDevice_->CreateBuffer(&bd_vertex, &data_vertex, &pVertexBuffer_);
126

249 bd.MiscFlags = 0;
250

254 InitData.SysMemSlicePitch = 0;
255 if (FAILED(Direct3D::pDevice_->CreateBuffer(&bd, &InitData, &ppIndexBuffer_[i])))
256 {
257 //MessageBox(0, "インデックスバッファの生成に失敗", fbxFileName, MB_OK);
258 //return FALSE;

268//骨の情報を準備
269void FbxParts::InitSkelton(FbxMesh * pMesh)
270{
271 // デフォーマ情報（ボーンとモデルの関連付け）の取得
272 FbxDeformer * pDeformer = pMesh->GetDeformer(0);
273 if (pDeformer == nullptr)
274 {
275 //ボーン情報なし
276 return;
277 }
278
279
280 // デフォーマ情報からスキンメッシュ情報を取得
281 pSkinInfo_ = (FbxSkin *)pDeformer;
282
283 // 頂点からポリゴンを逆引きするための情報を作成する
284 struct POLY_INDEX
285 {
286 int * polyIndex; // ポリゴンの番号
287 int * vertexIndex; // 頂点の番号
288 int numRef; // 頂点を共有するポリゴンの数
289 };
290
291 POLY_INDEX * polyTable = new POLY_INDEX[vertexCount_];
292 for (DWORD i = 0; i < vertexCount_; i++)
293 {
294 // 三角形ポリゴンに合わせて、頂点とポリゴンの関連情報を構築する
295 // 総頂点数＝ポリゴン数×３頂点
296 polyTable[i].polyIndex = new int[polygonCount_ * 3];
297 polyTable[i].vertexIndex = new int[polygonCount_ * 3];
298 polyTable[i].numRef = 0;
299 ZeroMemory(polyTable[i].polyIndex, sizeof(int)* polygonCount_ * 3);
300 ZeroMemory(polyTable[i].vertexIndex, sizeof(int)* polygonCount_ * 3);
301
302 // ポリゴン間で共有する頂点を列挙する
303 for (DWORD k = 0; k < polygonCount_; k++)
304 {
305 for (int m = 0; m < 3; m++)
306 {
307 if (pMesh->GetPolygonVertex(k, m) == i)
308 {
309 polyTable[i].polyIndex[polyTable[i].numRef] = k;
310 polyTable[i].vertexIndex[polyTable[i].numRef] = m;
311 polyTable[i].numRef++;
312 }
313 }
314 }
315 }
316
317 // ボーン情報を取得する
318 numBone_ = pSkinInfo_->GetClusterCount();
319 ppCluster_ = new FbxCluster*[numBone_];
320 for (int i = 0; i < numBone_; i++)
321 {
322 ppCluster_[i] = pSkinInfo_->GetCluster(i);
323 }
324
325 // ボーンの数に合わせてウェイト情報を準備する
326 pWeightArray_ = new FbxParts::Weight[vertexCount_];
327 for (DWORD i = 0; i < vertexCount_; i++)
328 {
329 pWeightArray_[i].posOrigin = pVertexData_[i].position;
330 pWeightArray_[i].normalOrigin = pVertexData_[i].normal;
331 pWeightArray_[i].pBoneIndex = new int[numBone_];
332 pWeightArray_[i].pBoneWeight = new float[numBone_];
333 for (int j = 0; j < numBone_; j++)
334 {
335 pWeightArray_[i].pBoneIndex[j] = -1;
336 pWeightArray_[i].pBoneWeight[j] = 0.0f;
337 }
338 }
339
340
341
342
343 // それぞれのボーンに影響を受ける頂点を調べる
344 // そこから逆に、頂点ベースでボーンインデックス・重みを整頓する
345 for (int i = 0; i < numBone_; i++)
346 {
347 int numIndex = ppCluster_[i]->GetControlPointIndicesCount(); //このボーンに影響を受ける頂点数
348 int * piIndex = ppCluster_[i]->GetControlPointIndices(); //ボーン/ウェイト情報の番号
349 double * pdWeight = ppCluster_[i]->GetControlPointWeights(); //頂点ごとのウェイト情報
350
351 //頂点側からインデックスをたどって、頂点サイドで整理する
352 for (int k = 0; k < numIndex; k++)
353 {
354 // 頂点に関連付けられたウェイト情報がボーン５本以上の場合は、重みの大きい順に４本に絞る
355 for (int m = 0; m < 4; m++)
356 {
357 if (m >= numBone_)
358 break;
359
360 if (pdWeight[k] > pWeightArray_[piIndex[k]].pBoneWeight[m])
361 {
362 for (int n = numBone_ - 1; n > m; n--)
363 {
364 pWeightArray_[piIndex[k]].pBoneIndex[n] = pWeightArray_[piIndex[k]].pBoneIndex[n - 1];
365 pWeightArray_[piIndex[k]].pBoneWeight[n] = pWeightArray_[piIndex[k]].pBoneWeight[n - 1];
366 }
367 pWeightArray_[piIndex[k]].pBoneIndex[m] = i;
368 pWeightArray_[piIndex[k]].pBoneWeight[m] = (float)pdWeight[k];
369 break;
370 }
371 }
372
373 }
374 }
375
376 //ボーンを生成
377 pBoneArray_ = new FbxParts::Bone[numBone_];
378 for (int i = 0; i < numBone_; i++)
379 {
380 // ボーンのデフォルト位置を取得する
381 FbxAMatrix matrix;
382 ppCluster_[i]->GetTransformLinkMatrix(matrix);
383
384 // 行列コピー（Fbx形式からDirectXへの変換）
385 XMFLOAT4X4 pose;
386 for (DWORD x = 0; x < 4; x++)
387 {
388 for (DWORD y = 0; y < 4; y++)
389 {
390 pose(x,y) = (float)matrix.Get(x, y);
391 }
392 }
393 pBoneArray_[i].bindPose = XMLoadFloat4x4(&pose);
394 }
395
396 // 一時的なメモリ領域を解放する
397 for (DWORD i = 0; i < vertexCount_; i++)
398 {
399 SAFE_DELETE_ARRAY(polyTable[i].polyIndex);
400 SAFE_DELETE_ARRAY(polyTable[i].vertexIndex);
401 }
402 SAFE_DELETE_ARRAY(polyTable);
403
404}

420//描画
421void FbxParts::Draw(Transform& transform)
422{
423 //今から描画する頂点情報をシェーダに伝える
424 UINT stride = sizeof(VERTEX);
425 UINT offset = 0;
426 Direct3D::pContext_->IASetVertexBuffers(0, 1, &pVertexBuffer_, &stride, &offset);
427
428 //使用するコンスタントバッファをシェーダに伝える
429 Direct3D::pContext_->VSSetConstantBuffers(0, 1, &pConstantBuffer_);
430 Direct3D::pContext_->PSSetConstantBuffers(0, 1, &pConstantBuffer_);
431
432
433 //シェーダーのコンスタントバッファーに各種データを渡す
434 for (DWORD i = 0; i < materialCount_; i++)
435 {
436 // インデックスバッファーをセット
437 UINT stride = sizeof(int);
438 UINT offset = 0;
439 Direct3D::pContext_->IASetIndexBuffer(ppIndexBuffer_[i], DXGI_FORMAT_R32_UINT, 0);
440
441
442 // パラメータの受け渡し
443 D3D11_MAPPED_SUBRESOURCE pdata;
444 CONSTANT_BUFFER cb;
445 cb.worldVewProj = XMMatrixTranspose(transform.GetWorldMatrix() * Camera::GetViewMatrix() * Camera::GetProjectionMatrix()); // リソースへ送る値をセット
446 cb.world = XMMatrixTranspose(transform.GetWorldMatrix());
447 cb.normalTrans = XMMatrixTranspose(transform.matRotate_ * XMMatrixInverse(nullptr, transform.matScale_));
448 cb.ambient = pMaterial_[i].ambient;
449 cb.diffuse = pMaterial_[i].diffuse;
450 cb.speculer = pMaterial_[i].specular;
451 cb.shininess = pMaterial_[i].shininess;
452 cb.cameraPosition = XMFLOAT4(Camera::GetPosition().x, Camera::GetPosition().y, Camera::GetPosition().z, 0);
453 cb.lightDirection = XMFLOAT4(1, -1, 1, 0);
454 cb.isTexture = pMaterial_[i].pTexture != nullptr;
455
456
457 Direct3D::pContext_->Map(pConstantBuffer_, 0, D3D11_MAP_WRITE_DISCARD, 0, &pdata); // GPUからのリソースアクセスを一時止める
458 memcpy_s(pdata.pData, pdata.RowPitch, (void*)(&cb), sizeof(cb)); // リソースへ値を送る
459
460
461
462 // テクスチャをシェーダーに設定
463
464 if (cb.isTexture)
465 {
466 ID3D11SamplerState* pSampler = pMaterial_[i].pTexture->GetSampler();
467 Direct3D::pContext_->PSSetSamplers(0, 1, &pSampler);
468
469 ID3D11ShaderResourceView* pSRV = pMaterial_[i].pTexture->GetSRV();
470 Direct3D::pContext_->PSSetShaderResources(0, 1, &pSRV);
471 }
472 Direct3D::pContext_->Unmap(pConstantBuffer_, 0); // GPUからのリソースアクセスを再開
473
474 //ポリゴンメッシュを描画する
475 Direct3D::pContext_->DrawIndexed(pMaterial_[i].polygonCount * 3, 0, 0);
476 }
477

480//ボーン有りのモデルを描画
481void FbxParts::DrawSkinAnime(Transform& transform, FbxTime time)
482{
483 // ボーンごとの現在の行列を取得する
484 for (int i = 0; i < numBone_; i++)
485 {
486 FbxAnimEvaluator * evaluator = ppCluster_[i]->GetLink()->GetScene()->GetAnimationEvaluator();
487 FbxMatrix mCurrentOrentation = evaluator->GetNodeGlobalTransform(ppCluster_[i]->GetLink(), time);
488
489 // 行列コピー（Fbx形式からDirectXへの変換）
490 XMFLOAT4X4 pose;
491 for (DWORD x = 0; x < 4; x++)
492 {
493 for (DWORD y = 0; y < 4; y++)
494 {
495 pose(x, y) = (float)mCurrentOrentation.Get(x, y);
496 }
497 }
498
499 // オフセット時のポーズの差分を計算する
500 pBoneArray_[i].newPose = XMLoadFloat4x4(&pose);
501 pBoneArray_[i].diffPose = XMMatrixInverse(nullptr, pBoneArray_[i].bindPose);
502 pBoneArray_[i].diffPose *= pBoneArray_[i].newPose;
503 }
504
505 // 各ボーンに対応した頂点の変形制御
506 for (DWORD i = 0; i < vertexCount_; i++)
507 {
508 // 各頂点ごとに、「影響するボーン×ウェイト値」を反映させた関節行列を作成する
509 XMMATRIX matrix;
510 ZeroMemory(&matrix, sizeof(matrix));
511 for (int m = 0; m < numBone_; m++)
512 {
513 if (pWeightArray_[i].pBoneIndex[m] < 0)
514 {
515 break;
516 }
517 matrix += pBoneArray_[pWeightArray_[i].pBoneIndex[m]].diffPose * pWeightArray_[i].pBoneWeight[m];
518
519 }
520
521 // 作成された関節行列を使って、頂点を変形する
522 XMVECTOR Pos = XMLoadFloat3(&pWeightArray_[i].posOrigin);
523 XMVECTOR Normal = XMLoadFloat3(&pWeightArray_[i].normalOrigin);
524 XMStoreFloat3(&pVertexData_[i].position,XMVector3TransformCoord(Pos, matrix));
525 XMStoreFloat3(&pVertexData_[i].normal, XMVector3TransformCoord(Normal, matrix));
526
527 }
528
529 // 頂点バッファをロックして、変形させた後の頂点情報で上書きする
530 D3D11_MAPPED_SUBRESOURCE msr = {};
531 Direct3D::pContext_->Map(pVertexBuffer_, 0, D3D11_MAP_WRITE_DISCARD, 0, &msr);
532 if (msr.pData)
533 {
534 memcpy_s(msr.pData, msr.RowPitch, pVertexData_, sizeof(VERTEX) * vertexCount_);
535 Direct3D::pContext_->Unmap(pVertexBuffer_, 0);
536 }
537
538
539 Draw(transform);

582void FbxParts::RayCast(RayCastData * data)
583{
584 data->hit = FALSE;
585
586 //マテリアル毎
587 for (DWORD i = 0; i < materialCount_; i++)
588 {
589 //そのマテリアルのポリゴン毎
590 for (DWORD j = 0; j < pMaterial_[i].polygonCount; j++)
591 {
592 //3頂点
593 XMFLOAT3 ver[3];
594 ver[0] = pVertexData_[ppIndexData_[i][j * 3 + 0]].position;
595 ver[1] = pVertexData_[ppIndexData_[i][j * 3 + 1]].position;
596 ver[2] = pVertexData_[ppIndexData_[i][j * 3 + 2]].position;
597
598 BOOL hit = FALSE;
