89{
90 auto Transition = [&](ModeratorSequence::s nextState) {
91 transitionTime -= dt;
92 if (transitionTime < 0)
93 state = nextState;
94 return (transitionTime < 0);
95 };
96 switch (state)
97 {
98 case CHANGED:
99

102

104 case PREP:

106
107 break;
108 case BEGIN:
109 if (true) {//時間条件、ポーズとかしてたら入らない
110 auto size = spawnDataList.at(waves).size();
111 if (spawnindex < size) {

113 float time = static_cast<float>(duration_cast<seconds>(curTime).count());
114 if (waiting.spawntime <= time)

119

121 else if (manager->Eliminated())

123 state = END;

128
129 }
130
131 break;
132 case END:
133 Transition(NEXT);
134
135 break;
136 case NEXT:
137 spawnindex = 0;
