104void Sprite::Draw(Transform& transform, RECT rect, float alpha)
105{
106 //いろいろ設定
107 Direct3D::SetShader(Direct3D::SHADER_2D);
108 UINT stride = sizeof(VERTEX);
109 UINT offset = 0;
110 Direct3D::pContext_->IASetVertexBuffers(0, 1, &pVertexBuffer_, &stride, &offset);
111 Direct3D::pContext_->VSSetConstantBuffers(0, 1, &pConstantBuffer_);
112 Direct3D::pContext_->PSSetConstantBuffers(0, 1, &pConstantBuffer_);
113 Direct3D::SetDepthBafferWriteEnable(false);
114
115
116 // インデックスバッファーをセット
117 stride = sizeof(int);
118 offset = 0;
119 Direct3D::pContext_->IASetIndexBuffer(pIndexBuffer_, DXGI_FORMAT_R32_UINT, 0);
120
121 // パラメータの受け渡し
122 D3D11_MAPPED_SUBRESOURCE pdata;
123 CONSTANT_BUFFER cb;
124
125
126 //表示するサイズに合わせる
127 XMMATRIX cut = XMMatrixScaling((float)rect.right, (float)rect.bottom ,1);
128
129 //画面に合わせる
130 XMMATRIX view = XMMatrixScaling(1.0f / Direct3D::screenWidth_, 1.0f / Direct3D::screenHeight_, 1.0f);
131
132 //最終的な行列
133 XMMATRIX world = cut * transform.matScale_ * transform.matRotate_ * view * transform.matTranslate_;
134 cb.world = XMMatrixTranspose(world);
135
136 auto texSize = pTexture_->GetExtendedSize();
137 // テクスチャ座標変換行列を渡す
138 XMMATRIX mTexTrans = XMMatrixTranslation((float)rect.left / (float)texSize.x,(float)rect.top / (float)texSize.y, 0.0f);
139 XMMATRIX mTexScale = XMMatrixScaling((float)rect.right / (float)texSize.x,(float)rect.bottom / (float)texSize.y, 1.0f);
140 XMMATRIX mTexel = mTexScale * mTexTrans;
141 cb.uvTrans = XMMatrixTranspose(mTexel);
142
143
144 // テクスチャ合成色情報を渡す
145 cb.color = XMFLOAT4(1, 1, 1, alpha);
146
147 Direct3D::pContext_->Map(pConstantBuffer_, 0, D3D11_MAP_WRITE_DISCARD, 0, &pdata); // GPUからのリソースアクセスを一時止める
148 memcpy_s(pdata.pData, pdata.RowPitch, (void*)(&cb), sizeof(cb)); // リソースへ値を送る
149
150
151 ID3D11SamplerState* pSampler = pTexture_->GetSampler();
152 Direct3D::pContext_->PSSetSamplers(0, 1, &pSampler);
153
154 ID3D11ShaderResourceView* pSRV = pTexture_->GetSRV();
155 Direct3D::pContext_->PSSetShaderResources(0, 1, &pSRV);
156
157 Direct3D::pContext_->Unmap(pConstantBuffer_, 0); // GPUからのリソースアクセスを再開
158
159 //ポリゴンメッシュを描画する
160 Direct3D::pContext_->DrawIndexed(6, 0, 0);
161
162 Direct3D::SetShader(Direct3D::SHADER_3D);
163
164 Direct3D::SetDepthBafferWriteEnable(true);
165
166}
