447 //三角形と線分の衝突判定（衝突判定に使用）
448 //https://pheema.hatenablog.jp/entry/ray-triangle-intersection
449 bool Intersect(XMFLOAT3 & start, XMFLOAT3 & direction, XMFLOAT3 & v0, XMFLOAT3 & v1, XMFLOAT3 & v2, float* distance)
450 {
451 // 微小な定数([M?ller97] での値)
452 constexpr float kEpsilon = 1e-6f;
453
454 //三角形の２辺
455 XMVECTOR edge1 = XMVectorSet(v1.x - v0.x, v1.y - v0.y, v1.z - v0.z, 0.0f);
456 XMVECTOR edge2 = XMVectorSet(v2.x - v0.x, v2.y - v0.y, v2.z - v0.z, 0.0f);
457
458 XMVECTOR alpha = XMVector3Cross(XMLoadFloat3(&direction), edge2);
459 float det = XMVector3Dot(edge1, alpha).m128_f32[0];
460
461 // 三角形に対して、レイが平行に入射するような場合 det = 0 となる。
462 // det が小さすぎると 1/det が大きくなりすぎて数値的に不安定になるので
463 // det ? 0 の場合は交差しないこととする。
464 if (-kEpsilon < det && det < kEpsilon)
465 {
466 return false;
467 }
468
469 float invDet = 1.0f / det;
470 XMFLOAT3 r = XMFLOAT3(start.x - v0.x, start.y - v0.y, start.z - v0.z);
471
472 // u が 0 <= u <= 1 を満たしているかを調べる。
473 float u = XMVector3Dot(alpha, XMLoadFloat3(&r)).m128_f32[0] * invDet;
474 if (u < 0.0f || u > 1.0f)
475 {
476 return false;
477 }
478
479 XMVECTOR beta = XMVector3Cross(XMLoadFloat3(&r), edge1);
480
481 // v が 0 <= v <= 1 かつ u + v <= 1 を満たすことを調べる。
482 // すなわち、v が 0 <= v <= 1 - u をみたしているかを調べればOK。
483 float v = XMVector3Dot(XMLoadFloat3(&direction), beta).m128_f32[0] * invDet;
484 if (v < 0.0f || u + v > 1.0f)
485 {
486 return false;
487 }
488
489 // t が 0 <= t を満たすことを調べる。
490 float t = XMVector3Dot(edge2, beta).m128_f32[0] * invDet;
491 if (t < 0.0f)
492 {
493 return false;
494 }
495
496 *distance = t;
497 return true;
498 }
